class ResidualBlock(nn.Module):
젨젨def __init__(self, channels):
젨젨젨젨super().__init__()

젨젨젨젨젨젨ConvBlock(channels, channels, kernel_size=3, padding=1),

젨젨def forward(self, x): return x + self.block(x)

class Generator(nn.Module):
젨젨def __init__(self, img_channels=3, num_features=64, num_residuals=9):
젨젨젨젨super().__init__()
젨젨젨젨self.initial = nn.Sequential(
젨젨젨젨젨젨nn.Conv2d(img_channels, num_features, kernel_size=7, stride=1, padding=3, padding_mode="reflect"),
젨젨젨젨젨젨nn.InstanceNorm2d(num_features),
젨젨젨젨젨젨nn.ReLU(inplace=True)
젨젨젨젨)
젨젨젨젨self.down_blocks = nn.ModuleList([
젨젨젨젨젨젨ConvBlock(num_features, num_features*2, kernel_size=3, stride=2, padding=1),
젨젨젨젨젨젨ConvBlock(num_features*2, num_features*4, kernel_size=3, stride=2, padding=1),
젨젨젨젨])
젨젨젨젨self.res_blocks = nn.Sequential(*[ResidualBlock(num_features*4) for _ in range(num_residuals)])
젨젨젨젨self.up_blocks = nn.ModuleList([
젨젨젨젨젨젨ConvBlock(num_features*4, num_features*2, down=False, kernel_size=3, stride=2, padding=1, output_padding=1),
젨젨젨젨젨젨ConvBlock(num_features*2, num_features, down=False, kernel_size=3, stride=2, padding=1, output_padding=1),
젨젨젨젨])
젨젨젨젨self.last = nn.Conv2d(num_features, img_channels, kernel_size=7, stride=1, padding=3, padding_mode="reflect")

젨젨def forward(self, x):
젨젨젨젨x = self.initial(x)
젨젨젨젨for layer in self.down_blocks: x = layer(x)
젨젨젨젨x = self.res_blocks(x)
젨젨젨젨for layer in self.up_blocks: x = layer(x)
젨젨젨젨return torch.tanh(self.last(x))

class Discriminator(nn.Module):
젨젨def __init__(self, in_channels=3, features=[64, 128, 256, 512]):
젨젨젨젨super().__init__()

젨젨젨젨젨젨nn.Conv2d(in_channels, features[0], kernel_size=4, stride=2, padding=1, padding_mode="reflect"),
젨젨젨젨젨젨nn.LeakyReLU(0.2, inplace=True)
젨젨젨젨)
젨젨젨젨layers = []
젨젨젨젨in_channels = features[0]
젨젨젨젨for feature in features[1:]:
젨젨젨젨젨젨layers.append(ConvBlock(in_channels, feature, stride=1 if feature==features[-1] else 2, kernel_size=4, padding=1))
젨젨젨젨젨젨in_channels = feature
젨젨젨젨layers.append(nn.Conv2d(in_channels, 1, kernel_size=4, stride=1, padding=1, padding_mode="reflect"))
젨젨젨젨self.model = nn.Sequential(self.initial, *layers)

젨젨def forward(self, x): return torch.sigmoid(self.model(x))

Cycle consistency loss ensures that when an image is translated from one domain to another and then translated back, it remains similar to the original image. This helps the model preserve the important content of the image while changing only its style. It prevents the generators from making random or unrealistic transformations. As a result, the generated images become more consistent and meaningful.
